Creation Database
    Modif fichier .env
        DB_USER=root
        DB_PASSWORD=
        DB_HOST=localhost
        DB_PORT=3306
        DB_NAME=symfonytest
        DATABASE_URL=mysql://${DB_USER}:${DB_PASSWORD}@${DB_HOST}:${DB_PORT}/${DB_NAME}
symfony console doctrine:database:create
